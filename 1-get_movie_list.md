```python
def onehot(df):
    categories = []
    lista = list(df['genres'].unique())
    for cat in lista:
        for c in cat.split(','):
            if c not in categories:
                if c != '\\N':
                    categories.append(c)
    for c in categories:
        df['genre_'+c] = df['genres'].apply(lambda x: 1 if c in x else 0)
    return df
```
